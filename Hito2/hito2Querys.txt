---------Atleta-------
create table if not exists atleta(
    id integer primary key,
    nombre varchar(255) not null,
    genero varchar(255) not null,
    nacionalidad varchar(255) not null,
    pais_residencia varchar(255),
    altura integer,
    peso integer,
    fecha_nacimiento varchar(255) not null,
    lugar_nacimiento varchar(255)
)

------Entrenador----
create table if not exists entrenador(
    id integer primary key,
    nombre varchar(255) not null,
    genero varchar(255) not null,
    funcion varchar(255) not null,
    categoria varchar(255) not null,
    nacionalidad varchar(255) not null
)

-----Pais------
create table if not exists pais(
    nombre varchar(255) primary key,
    recuento_medallas integer not null,




    oros integer not null,
    platas integer not null,
    bronces integer not null,
)

------Disciplina-------
create table if not exists disciplina(
    id serial primary key,
    nombre varchar(255) not null
)

--------Evento--------
create table if not exists evento(
    nombre varchar(255) primary key,
    ganador not null
)

-------Deporte--------
create table if not exists deporte(
    nombre varchar(255) not null
)

-------Medalla-------
create table if not exists medalla(
    id integer not null,
    evento_ganador integer not null,
    atleta_id integer not null,
    tipo varchar(255) not null,
    primary key (id, evento_ganador, atleta_id),
    foreign key (evento_ganador) references evento(ganador),
    foreign key (atleta_id) references atleta(id),
)
